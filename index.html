Actividad 3 - Ejercicios de SQL

        create database escuela;
        use escuela;

        create table nuevos_estudiantes(
	    id int primary key,
	    nombre varchar(255),
	    edad int
        );

        create table estudiantes_graduados(
	    id int primary key auto_increment,
	    nombre varchar(255),
	    edad int
        );

        create table clase_a(
	    id INT primary key,
	    nombre VARCHAR(255)
        );

        create table nueva_clase_b(
	    id INT primary key,
	    nombre VARCHAR(255),
	    clase_a_id INT,
	    foreign key (clase_a_id) references clase_a(id)
        );
